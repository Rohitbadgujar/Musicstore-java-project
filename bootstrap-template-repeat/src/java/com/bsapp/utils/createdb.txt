DROP TABLE "USERDATA";

CREATE TABLE "USERDATA"
(
"USER_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"EMAIL" VARCHAR(50), 
"PASSWORD" VARCHAR(50), 
"FNAME" VARCHAR(50),   
"LNAME" VARCHAR(50),
"USERTYPE" VARCHAR(10)
);

DROP TABLE "ALBUM";

CREATE TABLE "ALBUM"
(
"ALBUM_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"NAME" VARCHAR(50),
"GENRE" VARCHAR(4),
"ARTIST_NAME" VARCHAR(100), 
"LONG_DESCRIPTION" VARCHAR(100)
);
